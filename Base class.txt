package base;

import java.io.File;
import java.time.Duration;

import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.OutputType;
import org.openqa.selenium.TakesScreenshot;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import org.openqa.selenium.io.FileHandler;
import org.openqa.selenium.support.ui.WebDriverWait;

import com.proect.page.LoginRegisterPage;

import io.cucumber.java.AfterAll;
import io.cucumber.java.BeforeAll;

public class BaseClass {

	public static WebDriver driver;
	public static WebDriverWait wait;

	public static final String BASE_URL = "https://d2mfj34wvggqrd.cloudfront.net/";
	public static final String REGISTER_URL = "https://d2mfj34wvggqrd.cloudfront.net/register";
	public static final String SCREENSHOT_PATH = "screenshots/";
	
	
	public static void launchApplication(){

		driver = new ChromeDriver();
		driver.manage().window().maximize();
		wait = new WebDriverWait(driver, Duration.ofSeconds(5));
		System.out.println("Browser Launched Successfully");

	}

	public static void navigateTo(String url) throws InterruptedException {
		driver.get(url);
		Thread.sleep(2000);
		System.out.println("Navigated to: " + url);
	}


	public static void takeScreenshot(String fileName) {
		try {
			new File(SCREENSHOT_PATH).mkdirs();
			TakesScreenshot ts = (TakesScreenshot) driver;
			File source = ts.getScreenshotAs(OutputType.FILE);
			File destination = new File(SCREENSHOT_PATH + fileName + ".png");
			FileHandler.copy(source, destination);
			System.out.println("Screenshot saved: " + SCREENSHOT_PATH + fileName + ".png");
		} catch (Exception e) {
			System.out.println("Screenshot failed: " + e.getMessage());
		}
	}

	public static void quitBrowser() {
		if (driver != null) {
			driver.quit();
			System.out.println("Browser Quit");
		}
	}
		
}
