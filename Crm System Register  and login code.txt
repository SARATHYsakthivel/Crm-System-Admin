package stepDefinition;

import java.time.Duration;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.support.ui.WebDriverWait;
import org.testng.Assert;

import com.proect.page.LoginRegisterPage;

import base.BaseClass;
import io.cucumber.java.Before;
import io.cucumber.java.en.And;
import io.cucumber.java.en.Given;
import io.cucumber.java.en.Then;
import io.cucumber.java.en.When;

public class CrmSystem {

	private static WebDriver driver;
	public static LoginRegisterPage loginregisterpage;

	@Before

	@Given("user launches browser for registration")
	public void user_launches_browser_for_registration() {

	}

	@When("user opens CRM register page")
	public void user_opens_crm_register_page() throws InterruptedException {

		BaseClass.navigateTo(BaseClass.REGISTER_URL);

		WebDriverWait wait=new WebDriverWait(BaseClass.driver,Duration.ofSeconds(10));

	}

	@And("user enters username as {string}")
	public void user_enters_username(String name) {
		BaseClass.driver.findElement(loginregisterpage.username).sendKeys(name);
	}

	@And("user enters register email as {string}")
	public void user_enters_register_email(String email) {
		BaseClass.driver.findElement(loginregisterpage.email).sendKeys(email);

	}

	@And("user enters password as {string}")
	public void user_enters_password(String password) {
		BaseClass.driver.findElement(loginregisterpage.password).sendKeys(password);
	}

	@And("user enters confirm password as {string}")
	public void user_enters_confirm_password(String confirmPassword) {
		BaseClass.driver.findElement(loginregisterpage.confirmPassword).sendKeys(confirmPassword);
	}

	@And("user enters mobile number as {string}")
	public void user_enters_mobile_number(String mobile) {
		BaseClass.driver.findElement(loginregisterpage.mobileNumber).sendKeys(mobile);
	}

	@And("user clicks register button")
	public void user_clicks_register_button() throws InterruptedException {

		BaseClass.driver.findElement(loginregisterpage.registerButton).click();

		BaseClass.takeScreenshot("Register button");

		WebDriverWait wait=new WebDriverWait(BaseClass.driver,Duration.ofSeconds(10));
	}

	@Then("user should validate all the fields")
	public void user_should_validate_all_the_fields() {

		String currentUrl = BaseClass.driver.getCurrentUrl();
		Validation validator = new Validation(BaseClass.driver, loginregisterpage);
		if (currentUrl.contains("register")) {
			validator.validateAllFields();
		} else {
			System.out.println(" ");
		}
	System.out.println("Success");
	}

	@Given("user launches browser for login")
	public void user_launches_browser_for_login() {

	}

	@When("user opens CRM login page")
	public void user_opens_crm_login_page() throws InterruptedException {

		BaseClass.navigateTo(BaseClass.BASE_URL);

		WebDriverWait wait=new WebDriverWait(BaseClass.driver,Duration.ofSeconds(10));
	}

	@And("user enters login email as {string}")
	public void user_enters_login_email(String emailid) {

		BaseClass.driver.findElement(loginregisterpage.loginEmail).sendKeys(emailid);
	}

	@And("user enters login password as {string}")
	public void user_enters_login_password(String loginpassword) {

		BaseClass.driver.findElement(loginregisterpage.loginPassword).sendKeys(loginpassword);
	}

	@And("user clicks login button")
	public void user_clicks_login_button() throws InterruptedException {

		BaseClass.driver.findElement(loginregisterpage.loginButton).click();

		BaseClass.takeScreenshot("Login button clicked");

		Thread.sleep(3000);

	}

	@Then("user should validate all the field")
	public void user_should_validate_all_the_field() {

		String currentUrl = BaseClass.driver.getCurrentUrl();
		Validation validator = new Validation(BaseClass.driver, loginregisterpage);
		if (currentUrl.contains("login")) {
			validator.validateAllFields();
		} else {
			System.out.println("");
		}

	}

}
